function renderDashboard(records) { layersGroup.clearLayers(); let totalHighRisk = 0; let totalHours = 0; const feedContainer = document.getElementById('risk-feed'); feedContainer.innerHTML = ''; records.forEach(rec => { const riskScore = rec.biosecurityRiskScore || 0; const riskPct = Math.round(riskScore * 100); const isHighRisk = riskScore >= 0.70; if (isHighRisk) totalHighRisk++; totalHours += parseFloat(rec.residenceHours || 0); const primaryColor = isHighRisk ? '#ef4444' : '#f59e0b'; // 1. Draw Inter-Port Route Polyline (Departure -> Vessel Position -> Destination) if (rec.routeCoordinates && Array.isArray(rec.routeCoordinates)) { const validPoints = rec.routeCoordinates.filter(pt => pt && pt[0] !== null && pt[1] !== null); if (validPoints.length > 1) { const routeLine = L.polyline(validPoints, { color: primaryColor, weight: 2, opacity: 0.7, dashArray: '6, 8' // Dashed line style for active transit }); routeLine.bindTooltip(`
${rec.vesselName}: ${rec.portOfDeparture} → ${rec.portOfDestination}
`, { sticky: true }); layersGroup.addLayer(routeLine); } } // 2. Draw Vessel Marker if (rec.vesselPos && rec.vesselPos[0] !== null) { const vesselMarker = L.circleMarker(rec.vesselPos, { radius: isHighRisk ? 8 : 6, fillColor: primaryColor, color: '#ffffff', weight: 1.5, fillOpacity: 0.9 }); const popupContent = `
${rec.vesselName} (${rec.flag})
MMSI: ${rec.mmsi}
Origin Port: ${rec.portOfDeparture}
Current Port: ${rec.portName}
Destination Port: ${rec.portOfDestination}
Region: ${rec.region}
Residence Duration: ${rec.residenceHours} hrs
Biosecurity Risk: ${riskPct}%
`; vesselMarker.bindPopup(popupContent); layersGroup.addLayer(vesselMarker); } // 3. Render Sidebar Feed Card const card = document.createElement('div'); card.className = `p-3 rounded-lg border transition-all cursor-pointer hover:border-cyan-500 ${ isHighRisk ? 'bg-red-950/20 border-red-900/50' : 'bg-slate-800/40 border-slate-800' }`; card.innerHTML = `
${rec.vesselName} [${rec.flag}] ${riskPct}% Risk
Route: ${rec.portOfDeparture || 'Port'} → ${rec.portOfDestination || 'En Route'}
Region: ${rec.region}
In-Port Duration: ${rec.residenceHours} hrs
`; card.addEventListener('click', () => { if (rec.vesselPos && rec.vesselPos[0] !== null) { map.setView(rec.vesselPos, 6, { animate: true }); } }); feedContainer.appendChild(card); }); // Update KPI Bar document.getElementById('kpi-total-visits').innerText = records.length; document.getElementById('kpi-high-risk').innerText = totalHighRisk; document.getElementById('kpi-avg-residence').innerHTML = `${(totalHours / (records.length || 1)).toFixed(1)} hrs`; }
